1//! `run_if` condition evaluation for validator dispatch.
2//!
3//! Parses and evaluates expressions of the form `name.status == value`,
4//! joined by `and`/`or` with left-to-right evaluation and no short-circuit.
5//! Missing validators are treated as `skip`.
6
7use std::collections::BTreeMap;
8
9use crate::config::split_run_if;
10use crate::error::{BatonError, Result};
11use crate::types::*;
12
13/// Evaluate a run_if expression against prior validator results.
14///
15/// Parses `expr` as a sequence of `name.status == value` atoms joined by
16/// `and`/`or` operators. Evaluation is left-to-right with no precedence
17/// and no short-circuit (all atoms are evaluated to catch missing references).
18///
19/// Returns `Err` if the expression is empty or references a validator not
20/// present in `prior_results`.
21pub fn evaluate_run_if(
22    expr: &str,
23    prior_results: &BTreeMap<String, ValidatorResult>,
24) -> Result<bool> {
25    let tokens = split_run_if(expr);
26
27    if tokens.is_empty() {
28        return Err(BatonError::ValidationError(
29            "Empty run_if expression".into(),
30        ));
31    }
32
33    // Evaluate first atom
34    let mut current_result = evaluate_atom(&tokens[0], prior_results)?;
35
36    // Evaluate remaining atoms left-to-right, no precedence, NO short-circuit.
37    let mut i = 1;
38    while i < tokens.len() {
39        if i + 1 >= tokens.len() {
40            return Err(BatonError::ValidationError(format!(
41                "Invalid run_if expression: '{expr}'"
42            )));
43        }
44        let operator = &tokens[i];
45        let next_result = evaluate_atom(&tokens[i + 1], prior_results)?;
46
47        match operator.as_str() {
48            "and" => current_result = current_result && next_result,
49            "or" => current_result = current_result || next_result,
50            other => {
51                return Err(BatonError::ValidationError(format!(
52                    "Invalid operator in run_if: '{other}'. Expected 'and' or 'or'."
53                )));
54            }
55        }
56
57        i += 2;
58    }
59
60    Ok(current_result)
61}
62
63fn evaluate_atom(atom: &str, prior_results: &BTreeMap<String, ValidatorResult>) -> Result<bool> {
64    let parts: Vec<&str> = atom.split(".status == ").collect();
65    if parts.len() != 2 {
66        return Err(BatonError::ValidationError(format!(
67            "Invalid run_if expression: '{atom}'. Expected '<name>.status == <value>'"
68        )));
69    }
70
71    let validator_name = parts[0].trim();
72    let expected_status = parts[1].trim();
73
74    let expected: Status = expected_status.parse().map_err(|_| {
75        BatonError::ValidationError(format!("Invalid status in run_if: '{expected_status}'"))
76    })?;
77
78    match prior_results.get(validator_name) {
79        Some(result) => Ok(result.status == expected),
80        None => Ok(expected == Status::Skip),
81    }
82}

84#[cfg(test)]
85mod tests {
86    use super::*;
87    use crate::test_helpers as th;
88
89    // ─── run_if evaluation ───────────────────────────
90
91    #[test]
92    fn run_if_simple_pass() {
93        let results = th::prior_results();
94        assert!(evaluate_run_if("lint.status == pass", &results).unwrap());
95    }
96
97    #[test]
98    fn run_if_simple_fail() {
99        let results = th::prior_results();
100        assert!(!evaluate_run_if("lint.status == fail", &results).unwrap());
101    }
102
103    #[test]
104    fn run_if_and_both_true() {
105        let results = th::prior_results();
106        assert!(
107            !evaluate_run_if("lint.status == pass and typecheck.status == pass", &results).unwrap()
108        );
109    }
110
111    #[test]
112    fn run_if_or_one_true() {
113        let results = th::prior_results();
114        assert!(
115            evaluate_run_if("lint.status == fail or typecheck.status == fail", &results).unwrap()
116        );
117    }
118
119    #[test]
120    fn run_if_left_to_right_no_precedence() {
121        // "a or b and c" → "(a or b) and c"
122        let mut results = BTreeMap::new();
123        results.insert("a".into(), th::result("a", Status::Pass));
124        results.insert("b".into(), th::result("b", Status::Fail));
125        results.insert("c".into(), th::result("c", Status::Fail));
126
127        // a.pass or b.pass → true or false → true
128        // true and c.pass → true and false → false
129        let result = evaluate_run_if(
130            "a.status == pass or b.status == pass and c.status == pass",
131            &results,
132        )
133        .unwrap();
134        assert!(!result);
135    }
136
137    #[test]
138    fn run_if_skipped_validator() {
139        let mut results = BTreeMap::new();
140        results.insert("a".into(), th::result("a", Status::Skip));
141        assert!(evaluate_run_if("a.status == skip", &results).unwrap());
142    }
143
144    #[test]
145    fn run_if_nonexistent_treated_as_skip() {
146        let results = BTreeMap::new();
147        assert!(evaluate_run_if("nonexistent.status == skip", &results).unwrap());
148        assert!(!evaluate_run_if("nonexistent.status == pass", &results).unwrap());
149    }
150
151    #[test]
152    fn run_if_invalid_expression() {
153        let results = BTreeMap::new();
154        assert!(evaluate_run_if("invalid expression", &results).is_err());
155    }
156
157    #[test]
158    fn run_if_empty_expression_returns_err() {
159        let results = BTreeMap::new();
160        let err = evaluate_run_if("", &results).unwrap_err();
161        assert!(
162            err.to_string().contains("Empty run_if"),
163            "expected 'Empty run_if' in error, got: {err}"
164        );
165    }
166
167    #[test]
168    fn run_if_unrecognized_status_returns_err() {
169        let results = th::prior_results();
170        let err = evaluate_run_if("lint.status == invalid_status", &results).unwrap_err();
171        assert!(
172            err.to_string().contains("Invalid status"),
173            "expected 'Invalid status' in error, got: {err}"
174        );
175    }
176
177    #[test]
178    fn run_if_expression_ending_with_operator_returns_err() {
179        let results = th::prior_results();
180        // "lint.status == pass and" — the trailing "and" gets absorbed into
181        // the atom text by split_run_if, producing an invalid atom that cannot
182        // be parsed. Either way an Err is returned.
183        let err = evaluate_run_if("lint.status == pass and", &results).unwrap_err();
184        let msg = err.to_string();
185        assert!(
186            msg.contains("Invalid"),
187            "expected 'Invalid' in error, got: {msg}"
188        );
189    }
