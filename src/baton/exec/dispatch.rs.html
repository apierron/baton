1//! Dispatch planner: maps validators + file pools to concrete [`Invocation`]s.
2//!
3//! Implements the four input declaration modes — `None` (single call, no files),
4//! `PerFile` (one call per matching file), `Batch` (one call with all matches),
5//! and `Named` (group files by a key expression such as `{stem}`).
6//! Returns `(invocations, warnings)`; empty invocations + warnings means skip.
7
8use std::collections::{BTreeMap, HashSet};
9use std::path::PathBuf;
10
11use crate::config::{InputDecl, ValidatorConfig};
12use crate::types::*;
13
14/// Match a filename against a glob pattern (e.g. "*.py", "src/**/*.rs").
15fn glob_matches(pattern: &str, path: &std::path::Path) -> bool {
16    let filename = path.file_name().unwrap_or_default().to_string_lossy();
17    let path_str = path.to_string_lossy();
18    // Try matching against full path first, then filename only
19    glob_match::glob_match(pattern, &path_str) || glob_match::glob_match(pattern, &filename)
20}
21
22/// Extract a key value from a file path based on a key expression.
23///
24/// Supported expressions: `{stem}`, `{name}`, `{ext}`.
25fn extract_key(expr: &str, path: &std::path::Path) -> Option<String> {
26    match expr {
27        "{stem}" => path.file_stem().map(|s| s.to_string_lossy().to_string()),
28        "{name}" => path.file_name().map(|s| s.to_string_lossy().to_string()),
29        "{ext}" => path.extension().map(|s| s.to_string_lossy().to_string()),
30        _ => Option::None,
31    }
32}

34/// Plan dispatch: turn a validator's input declaration + file pool into invocations.
35///
36/// Returns `(invocations, warnings)`. An empty invocations vec with warnings
37/// means the validator should be skipped.
38pub fn plan_dispatch(
39    validator: &ValidatorConfig,
40    pool: &[InputFile],
41) -> (Vec<Invocation>, Vec<String>) {
42    let mut warnings = Vec::new();
43
44    match &validator.input {
45        InputDecl::None => {
46            // DP-001: single invocation, no files
47            let inv = Invocation {
48                validator_name: validator.name.clone(),
49                group_key: Option::None,
50                inputs: BTreeMap::new(),
51            };
52            (vec![inv], warnings)
53        }
54        InputDecl::PerFile { pattern } => {
55            // DP-002: one invocation per matching file
56            let matching: Vec<&InputFile> = pool
57                .iter()
58                .filter(|f| glob_matches(pattern, &f.path))
59                .collect();
60
61            if matching.is_empty() {
62                // DP-007
63                warnings.push(format!(
64                    "Validator '{}': no files match pattern '{}'",
65                    validator.name, pattern
66                ));
67                return (vec![], warnings);
68            }
69
70            let invocations = matching
71                .into_iter()
72                .map(|f| {
73                    let mut inputs = BTreeMap::new();
74                    inputs.insert("file".to_string(), vec![f.clone()]);
75                    Invocation {
76                        validator_name: validator.name.clone(),
77                        group_key: Option::None,
78                        inputs,
79                    }
80                })
81                .collect();
82            (invocations, warnings)
83        }
84        InputDecl::Batch { pattern } => {
85            // DP-003: single invocation with all matching files
86            let matching: Vec<InputFile> = pool
87                .iter()
88                .filter(|f| glob_matches(pattern, &f.path))
89                .cloned()
90                .collect();
91
92            if matching.is_empty() {
93                warnings.push(format!(
94                    "Validator '{}': no files match pattern '{}'",
95                    validator.name, pattern
96                ));
97                return (vec![], warnings);
98            }
99
100            let mut inputs = BTreeMap::new();
101            inputs.insert("file".to_string(), matching);
102            let inv = Invocation {
103                validator_name: validator.name.clone(),
104                group_key: Option::None,
105                inputs,
106            };
107            (vec![inv], warnings)
108        }
109        InputDecl::Named(slots) => {
110            // Separate fixed inputs from glob-matched inputs
111            let mut fixed_inputs: BTreeMap<String, Vec<InputFile>> = BTreeMap::new();
112            let mut keyed_slots: Vec<(&String, &crate::config::InputSlot)> = Vec::new();
113            let mut unkeyed_slots: Vec<(&String, &crate::config::InputSlot)> = Vec::new();
114
115            for (slot_name, slot) in slots {
116                if let Some(ref path) = slot.path {
117                    // DP-006: fixed input
118                    fixed_inputs
119                        .insert(slot_name.clone(), vec![InputFile::new(PathBuf::from(path))]);
120                } else if slot.key.is_some() {
121                    keyed_slots.push((slot_name, slot));
122                } else {
123                    unkeyed_slots.push((slot_name, slot));
124                }
125            }
126
127            if keyed_slots.is_empty() && unkeyed_slots.is_empty() {
128                // Only fixed inputs — single invocation
129                let inv = Invocation {
130                    validator_name: validator.name.clone(),
131                    group_key: Option::None,
132                    inputs: fixed_inputs,
133                };
134                return (vec![inv], warnings);
135            }
136
137            if !keyed_slots.is_empty() {
138                // DP-004: group by key
139                // For each keyed slot, match files and extract keys
140                let mut slot_keys: BTreeMap<String, BTreeMap<String, Vec<InputFile>>> =
141                    BTreeMap::new(); // key_value -> slot_name -> files
142
143                for (slot_name, slot) in &keyed_slots {
144                    let pattern = slot.match_pattern.as_deref().unwrap_or("*");
145                    // baton-allow: unwrap — keyed_slots only contains items where slot.key.is_some()
146                    let key_expr = slot.key.as_deref().unwrap();
147
148                    for file in pool {
149                        if glob_matches(pattern, &file.path) {
150                            if let Some(key_val) = extract_key(key_expr, &file.path) {
151                                slot_keys
152                                    .entry(key_val)
153                                    .or_default()
154                                    .entry((*slot_name).clone())
155                                    .or_default()
156                                    .push(file.clone());
157                            }
158                        }
159                    }
160                }
161
162                // Collect all key values
163                let all_keys: HashSet<String> = slot_keys.keys().cloned().collect();
164                let slot_names: Vec<&String> = keyed_slots.iter().map(|(name, _)| *name).collect();
165
166                let mut invocations = Vec::new();
167                for key_val in &all_keys {
168                    // DP-005: check completeness
169                    let group = slot_keys.get(key_val);
170                    let complete = slot_names
171                        .iter()
172                        .all(|name| group.map(|g| g.contains_key(*name)).unwrap_or(false));
173
174                    if !complete {
175                        warnings.push(format!(
176                            "Validator '{}': incomplete group for key '{}', skipping",
177                            validator.name, key_val
178                        ));
179                        continue;
180                    }
181
182                    let mut inputs = fixed_inputs.clone();
183                    if let Some(group) = group {
184                        for (slot_name, files) in group {
185                            inputs.insert(slot_name.clone(), files.clone());
186                        }
187                    }
188
189                    // Also add unkeyed slots
190                    for (slot_name, slot) in &unkeyed_slots {
191                        let pattern = slot.match_pattern.as_deref().unwrap_or("*");
192                        let matching: Vec<InputFile> = pool
193                            .iter()
194                            .filter(|f| glob_matches(pattern, &f.path))
195                            .cloned()
196                            .collect();
197                        if !matching.is_empty() {
198                            if slot.collect {
199                                inputs.insert((*slot_name).clone(), matching);
200                            } else if let Some(first) = matching.into_iter().next() {
201                                inputs.insert((*slot_name).clone(), vec![first]);
202                            }
203                        }
204                    }
205
206                    invocations.push(Invocation {
207                        validator_name: validator.name.clone(),
208                        group_key: Some(key_val.clone()),
209                        inputs,
210                    });
211                }
212
213                if invocations.is_empty() && !all_keys.is_empty() {
214                    warnings.push(format!(
215                        "Validator '{}': all groups incomplete",
216                        validator.name
217                    ));
218                }
219                if all_keys.is_empty() {
220                    warnings.push(format!(
221                        "Validator '{}': no files match any keyed input patterns",
222                        validator.name
223                    ));
224                }
225
226                return (invocations, warnings);
227            }
228
229            // Only unkeyed named slots — single invocation
230            let mut inputs = fixed_inputs;
231            for (slot_name, slot) in &unkeyed_slots {
232                let pattern = slot.match_pattern.as_deref().unwrap_or("*");
233                let matching: Vec<InputFile> = pool
234                    .iter()
235                    .filter(|f| glob_matches(pattern, &f.path))
236                    .cloned()
237                    .collect();
238                if slot.collect {
239                    inputs.insert((*slot_name).clone(), matching);
240                } else if let Some(first) = matching.into_iter().next() {
241                    inputs.insert((*slot_name).clone(), vec![first]);
242                }
243            }
244
245            let inv = Invocation {
246                validator_name: validator.name.clone(),
247                group_key: Option::None,
248                inputs,
249            };
250            (vec![inv], warnings)
251        }
252    }
253}

274    fn dispatch_no_input_produces_single_invocation() {
275        // SPEC-EX-DP-001: validator with no input field produces one invocation
276        let v = validator_with_input("lint", crate::config::InputDecl::None);
277        let pool: Vec<InputFile> = vec![];
278        let (invocations, warnings) = plan_dispatch(&v, &pool);

287        // SPEC-EX-DP-002: per-file input produces one invocation per matching file
288        let v = validator_with_input(

307            std::path::PathBuf::from("/tmp/b.py")

331    #[test]
332    fn dispatch_keyed_inputs_grouped_by_key() {
333        // SPEC-EX-DP-004: named inputs with key expressions grouped by key value
334        use crate::config::InputSlot;
335
336        let mut slots = BTreeMap::new();
337        slots.insert(
338            "code".into(),
339            InputSlot {
340                match_pattern: Some("*.py".into()),
341                path: None,
342                key: Some("{stem}".into()),
343                collect: false,
344            },
345        );
346        slots.insert(
347            "spec".into(),
348            InputSlot {
349                match_pattern: Some("*.md".into()),
350                path: None,
351                key: Some("{stem}".into()),
352                collect: false,
353            },
354        );
355
356        let v = validator_with_input("check", crate::config::InputDecl::Named(slots));
357        let pool = vec![
358            InputFile::new(std::path::PathBuf::from("/tmp/a.py")),
359            InputFile::new(std::path::PathBuf::from("/tmp/b.py")),
360            InputFile::new(std::path::PathBuf::from("/tmp/a.md")),
361            InputFile::new(std::path::PathBuf::from("/tmp/b.md")),
362        ];
363
364        let (mut invocations, warnings) = plan_dispatch(&v, &pool);
365        invocations.sort_by(|a, b| a.group_key.cmp(&b.group_key));
366
367        assert_eq!(invocations.len(), 2);
368        assert_eq!(invocations[0].group_key, Some("a".into()));
369        assert_eq!(invocations[1].group_key, Some("b".into()));
370        assert_eq!(invocations[0].inputs.len(), 2); // code + spec
371        assert!(warnings.is_empty());
372    }
