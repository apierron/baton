1//! Initialize a new baton project with config and starter prompts.
2//!
3//! Creates `baton.toml`, the `.baton/` directory structure, and optionally
4//! starter prompt templates in `prompts/`. Supports interactive, flag-driven,
5//! and non-TTY (defaults) modes.
6
7// Config templates embedded from defaults/ at compile time.
8const DEFAULT_BASE_CONFIG: &str = include_str!("../../defaults/configs/base.toml");
9const DEFAULT_GENERIC_CONFIG: &str = include_str!("../../defaults/configs/generic.toml");
10const DEFAULT_RUST_CONFIG: &str = include_str!("../../defaults/configs/rust.toml");
11const DEFAULT_PYTHON_CONFIG: &str = include_str!("../../defaults/configs/python.toml");
12
13const DEFAULT_PROMPT_SPEC: &str = include_str!("../../defaults/prompts/spec-compliance.md");
14const DEFAULT_PROMPT_ADVERSARIAL: &str =
15    include_str!("../../defaults/prompts/adversarial-review.md");
16const DEFAULT_PROMPT_DOC: &str = include_str!("../../defaults/prompts/doc-completeness.md");
17
18const VALID_PROFILES: &[&str] = &["generic", "rust", "python"];
19
20/// Choices resolved from flags or interactive prompts.
21struct InitChoices {
22    include_code_validators: bool,
23    profile: String,
24    include_prompts: bool,
25}
26
27/// Run interactive prompts to determine init choices.
28fn prompt_init_choices() -> std::result::Result<InitChoices, String> {
29    use dialoguer::{Confirm, Select};
30
31    let include_code = Confirm::new()
32        .with_prompt("Include code validators?")
33        .default(true)
34        .interact()
35        .map_err(|e| format!("Interactive prompt failed: {e}"))?;
36
37    let profile = if include_code {
38        let options = &["Rust", "Python", "Generic"];
39        let idx = Select::new()
40            .with_prompt("Which language?")
41            .items(options)
42            .default(0)
43            .interact()
44            .map_err(|e| format!("Interactive prompt failed: {e}"))?;
45        match idx {
46            0 => "rust",
47            1 => "python",
48            _ => "generic",
49        }
50        .to_string()
51    } else {
52        String::new()
53    };
54
55    let include_prompts = Confirm::new()
56        .with_prompt("Include starter prompt templates?")
57        .default(true)
58        .interact()
59        .map_err(|e| format!("Interactive prompt failed: {e}"))?;
60
61    Ok(InitChoices {
62        include_code_validators: include_code,
63        profile,
64        include_prompts,
65    })
66}

68/// Initializes a new baton project: creates `baton.toml`, `.baton/` directory,
69/// and optionally starter prompt templates in `prompts/`.
70pub fn cmd_init(minimal: bool, prompts_only: bool, profile: Option<&str>) -> i32 {
71    // Validate explicit profile if provided
72    if let Some(p) = profile {
73        if !VALID_PROFILES.contains(&p) {
74            eprintln!(
75                "Error: unknown profile \"{p}\". Valid profiles: {}",
76                VALID_PROFILES.join(", ")
77            );
78            return 1;
79        }
80    }
81
82    // Resolve choices: explicit flags, interactive, or defaults
83    let choices = if profile.is_some() || minimal || prompts_only {
84        // Explicit flags — use them directly
85        InitChoices {
86            include_code_validators: true,
87            profile: profile.unwrap_or("generic").to_string(),
88            include_prompts: !minimal,
89        }
90    } else if atty::is(atty::Stream::Stdin) {
91        // Interactive mode
92        match prompt_init_choices() {
93            Ok(c) => c,
94            Err(e) => {
95                eprintln!("Error: {e}");
96                return 1;
97            }
98        }
99    } else {
100        // Non-TTY, no flags — use defaults
101        InitChoices {
102            include_code_validators: true,
103            profile: "generic".to_string(),
104            include_prompts: true,
105        }
106    };
107
108    if !prompts_only {
109        // Check if baton.toml already exists
110        if std::path::Path::new("baton.toml").exists() {
111            eprintln!("Error: baton.toml already exists. Will not overwrite.");
112            return 1;
113        }
114
115        // Create .baton directory structure
116        let baton_dir = std::path::Path::new(".baton");
117        if baton_dir.exists() {
118            eprintln!("Warning: .baton/ already exists. Creating missing subdirectories.");
119        }
120        for subdir in &["logs", "tmp"] {
121            let dir = baton_dir.join(subdir);
122            if let Err(e) = std::fs::create_dir_all(&dir) {
123                eprintln!("Error creating {}: {e}", dir.display());
124                return 1;
125            }
126        }
127
128        // Build config content
129        let starter_config = if choices.include_code_validators {
130            let profile_config = match choices.profile.as_str() {
131                "rust" => DEFAULT_RUST_CONFIG,
132                "python" => DEFAULT_PYTHON_CONFIG,
133                _ => DEFAULT_GENERIC_CONFIG,
134            };
135            format!("{}\n{}", DEFAULT_BASE_CONFIG, profile_config)
136        } else {
137            DEFAULT_BASE_CONFIG.to_string()
138        };
139
140        if let Err(e) = std::fs::write("baton.toml", starter_config) {
141            eprintln!("Error writing baton.toml: {e}");
142            return 1;
143        }
144        if choices.include_code_validators {
145            eprintln!("Created baton.toml (profile: {})", choices.profile);
146        } else {
147            eprintln!("Created baton.toml (base only, no validators)");
148        }
149        eprintln!("Created .baton/");
150    }
151
152    if choices.include_prompts || prompts_only {
153        // Create prompts directory with starter templates
154        let prompts_dir = std::path::Path::new("prompts");
155        if let Err(e) = std::fs::create_dir_all(prompts_dir) {
156            eprintln!("Error creating prompts/: {e}");
157            return 1;
158        }
159
160        let templates = [
161            ("spec-compliance.md", DEFAULT_PROMPT_SPEC),
162            ("adversarial-review.md", DEFAULT_PROMPT_ADVERSARIAL),
163            ("doc-completeness.md", DEFAULT_PROMPT_DOC),
164        ];
165
166        for (name, content) in &templates {
167            let path = prompts_dir.join(name);
168            if !path.exists() {
169                if let Err(e) = std::fs::write(&path, content) {
170                    eprintln!("Error writing {}: {e}", path.display());
171                    return 1;
172                }
173                eprintln!("Created prompts/{name}");
174            }
175        }
176    }
177
178    eprintln!("baton project initialized.");
179    0
180}
